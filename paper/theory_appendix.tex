\section{Analytical framework}

Let
\[
I[f]=\int_{-1}^{1}f(x)\,dx,\qquad
L_b(x)=\frac{1}{x^2+b^2}.
\]
With \(s=b^2\) and \(y=x^2\),
\[
F(s)=I[L_{\sqrt{s}}]
=\int_0^1\frac{y^{-1/2}}{s+y}\,dy
=\frac{2}{\sqrt{s}}\arctan\frac{1}{\sqrt{s}}.
\]

A symmetric \(n\)-pair rule is
\[
Q[f]=\sum_{j=1}^{n}w_j(f(-x_j)+f(x_j)),
\qquad 0<x_1<\cdots<x_n<1,\quad w_j>0.
\]
Writing \(y_j=x_j^2\),
\[
r_Q(s)=2\sum_{j=1}^{n}\frac{w_j}{s+y_j}.
\]

\begin{definition}
For \(0\le k\le n\), let \(\mathcal A_{n,k}\) be the set of such rules
satisfying
\[
\sum_{j=1}^{n}w_jy_j^\ell=\frac{1}{2\ell+1},
\qquad \ell=0,\ldots,k-1.
\]
\end{definition}

\begin{lemma}
A symmetric rule belongs to \(\mathcal A_{n,k}\) if and only if it is
exact on \(\mathbb P_{2k-1}\).
\end{lemma}

\begin{lemma}
A rule belongs to \(\mathcal A_{n,k}\) if and only if
\[
r_Q(s)-F(s)=O(s^{-k-1})
\qquad (s\to\infty).
\]
\end{lemma}

\begin{proposition}
If \(r_1,r_2\in\mathcal A_{n,k}\), then
\[
r_1-r_2=\frac{H}{q_1q_2},
\qquad
\deg H\le2n-k-1,
\]
unless \(H\equiv0\).
\end{proposition}

\begin{theorem}[Alternation certificate]
Let \(Q^\star\in\mathcal A_{n,k}\) and
\[
\varepsilon_Q(s)=\frac{r_Q(s)}{F(s)}-1.
\]
If \(\|\varepsilon_{Q^\star}\|_\infty=E\) and there exist \(2n-k+1\)
ordered points \(s_i\) such that
\[
\varepsilon_{Q^\star}(s_i)=\theta(-1)^iE,
\qquad \theta\in\{-1,1\},
\]
then no \(Q\in\mathcal A_{n,k}\) has strictly smaller uniform relative
error.
\end{theorem}

For a fixed rule \(Q\), define
\[
\Delta_j(Q)=I[T_j]-Q[T_j]
\]
and
\[
\mathcal H(C,\rho)=
\left\{
\sum_{j=0}^{\infty}a_jT_j:
|a_j|\le C\rho^{-j}
\right\},
\qquad \rho>1.
\]

\begin{theorem}[Exact coefficient-box norm]
\[
\sup_{h\in\mathcal H(C,\rho)}
|I[h]-Q[h]|
=
C\sum_{j=0}^{\infty}\rho^{-j}|\Delta_j(Q)|.
\]
\end{theorem}

Define
\[
S_Q=\max_{b\in B}|I[L_b]-Q[L_b]|.
\]

\begin{theorem}[Exact mixed-class risk]
For
\[
f=h+cL_b,\qquad
h\in\mathcal H(C,\rho),\quad
|c|\le c_{\max},\quad b\in B,
\]
\[
\sup_f|I[f]-Q[f]|
=
C G_Q(\rho)+c_{\max}S_Q.
\]
\end{theorem}

\paragraph{Rounded rules.}
The degree-reduction and alternation results require exact membership in
\(\mathcal A_{n,k}\). Published decimal rules are treated as fixed
rounded operators: their coefficient-box risks remain exact, while
their low-degree defects are retained in validated numerical
certification.
